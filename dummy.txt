// ── HELPER: extract JSON robustly ─────────────────────────────────────────────
function extractJSON(text) {
  if (!text || !text.trim()) throw new Error('Empty response from Gemini');
  try { return JSON.parse(text.trim()); } catch (_) {}
  const fenced = text.match(/```(?:json)?\s*([\s\S]*?)```/);
  if (fenced) { try { return JSON.parse(fenced[1].trim()); } catch (_) {} }
  const start = text.indexOf('{');
  const end   = text.lastIndexOf('}');
  if (start !== -1 && end !== -1 && end > start) {
    try { return JSON.parse(text.slice(start, end + 1)); } catch (_) {}
  }
  throw new Error(`Could not parse JSON from response: "${text.slice(0, 200)}"`);
}

// ── STAGE 1: PARSE SCRIPT (Gemini) ───────────────────────────────────────────
app.post('/api/parse', async (req, res) => {
  const { prompt, numShots } = req.body;
  if (!prompt) return res.status(400).json({ error: 'prompt is required' });

  const n = parseInt(numShots) || 4;

  console.log(`\n[parse] ── New request ──`);
  console.log(`[parse] Prompt: "${prompt.slice(0, 100)}"`);
  console.log(`[parse] Shots:  ${n}`);

  const systemInstruction = `You are a video production AI. Given a user's video idea, expand it into a structured shot list.
You MUST respond with ONLY a valid JSON object. No prose before it, no explanation after it, no markdown code fences.

Return exactly this JSON structure:
{
  "title": "short project title",
  "style": "cinematic visual style description",
  "character": "detailed character/subject description for consistency across all shots",
  "shots": [
    {
      "index": 1,
      "title": "short shot title",
      "description": "detailed visual description of this shot",
      "camera": "camera angle and movement",
      "duration": 5,
      "dialogue": false,
      "imagePrompt": "detailed image generation prompt including subject appearance, setting, lighting, mood, color palette, photographic style"
    }
  ]
}

Generate exactly ${n} shots. Each imagePrompt must be self-contained, cinematic, and vivid.`;

  try {
    console.log(`[parse] Calling Gemini API...`);

    const geminiRes = await fetch(
      `https://generativelanguage.googleapis.com/v1beta/models/gemini-2.5-flash:generateContent?key=${GOOGLE_KEY}`,
      {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({
          systemInstruction: { parts: [{ text: systemInstruction }] },
          contents: [{ role: 'user', parts: [{ text: prompt }] }],
          generationConfig: {
            temperature: 0.7,
            maxOutputTokens: 2500,
            responseMimeType: 'application/json'
          }
        })
      }
    );

    console.log(`[parse] Gemini HTTP status: ${geminiRes.status}`);

    if (!geminiRes.ok) {
      const errText = await geminiRes.text();
      console.error(`[parse] Gemini error: ${errText.slice(0, 400)}`);
      return res.status(geminiRes.status).json({
        error: `Gemini API returned ${geminiRes.status}: ${errText.slice(0, 300)}`
      });
    }

    const geminiData = await geminiRes.json();

    if (geminiData.error) {
      console.error(`[parse] Gemini API-level error:`, geminiData.error);
      return res.status(500).json({ error: geminiData.error.message });
    }

    const rawText = geminiData.candidates?.[0]?.content?.parts?.[0]?.text;

    if (!rawText) {
      console.error(`[parse] No text in response:`, JSON.stringify(geminiData).slice(0, 300));
      return res.status(500).json({ error: 'No text content in Gemini response' });
    }

    console.log(`[parse] Raw response (first 300):\n${rawText.slice(0, 300)}`);

    const parsed = extractJSON(rawText);

    if (!Array.isArray(parsed.shots)) {
      return res.status(500).json({ error: 'Response missing shots array — try again' });
    }

    console.log(`[parse] ✅ Success — ${parsed.shots.length} shots parsed`);
    return res.json(parsed);

  } catch (err) {
    console.error(`[parse] ❌ ${err.message}`);
    return res.status(500).json({ error: err.message });
  }
});

// ── STAGE 2: GENERATE IMAGE (Freepik Mystic) ──────────────────────────────────
app.post('/api/generate-image', async (req, res) => {
  const { prompt } = req.body;
  if (!prompt) return res.status(400).json({ error: 'prompt is required' });

  console.log(`\n[image] Generating via Freepik Mystic: "${prompt.slice(0, 80)}..."`);

  try {
    // Step 1 — Submit task
    const createRes = await fetch('https://api.freepik.com/v1/ai/mystic', {
      method: 'POST',
      headers: {
        'Content-Type': 'application/json',
        'x-freepik-api-key': FREEPIK_KEY
      },
      body: JSON.stringify({
        prompt:       prompt,
        aspect_ratio: 'widescreen_16_9',
        model:        'realism',
        resolution:   '2k',
        engine:       'automatic',
        filter_nsfw:  true
      })
    });

    console.log(`[image] Freepik create status: ${createRes.status}`);

    const createText = await createRes.text();
    if (!createRes.ok) {
      console.error(`[image] Freepik create error: ${createText.slice(0, 400)}`);
      return res.status(createRes.status).json({
        error: `Freepik API returned ${createRes.status}: ${createText.slice(0, 300)}`
      });
    }

    const createData = JSON.parse(createText);
    const taskId = createData.data?.task_id;

    if (!taskId) {
      console.error(`[image] No task_id in response:`, createText.slice(0, 300));
      return res.status(500).json({ error: 'No task_id returned from Freepik' });
    }

    console.log(`[image] Task created: ${taskId} — polling...`);

    // Step 2 — Poll until COMPLETED (max 20 attempts × 3s = 60s)
    for (let attempt = 0; attempt < 20; attempt++) {
      await new Promise(r => setTimeout(r, 3000));

      const pollRes  = await fetch(`https://api.freepik.com/v1/ai/mystic/${taskId}`, {
        headers: { 'x-freepik-api-key': FREEPIK_KEY }
      });
      const pollData = await pollRes.json();
      const status   = pollData.data?.status;

      console.log(`[image] Poll ${attempt + 1}: ${status}`);

      if (status === 'COMPLETED') {
        const imageUrl = pollData.data?.generated?.[0];
        if (!imageUrl) {
          return res.status(500).json({ error: 'Task completed but no image URL returned' });
        }
        const imgBuffer = await (await fetch(imageUrl)).buffer();
        const b64       = imgBuffer.toString('base64');
        console.log(`[image] ✅ Image ready`);
        return res.json({ imageBase64: b64, mimeType: 'image/jpeg' });
      }

      if (status === 'FAILED') {
        return res.status(500).json({ error: 'Freepik image task failed' });
      }
    }

    return res.status(500).json({ error: 'Freepik image task timed out after 60s' });

  } catch (err) {
    console.error(`[image] ❌ ${err.message}`);
    return res.status(500).json({ error: err.message });
  }
});

// ── STAGE 3: GENERATE SINGLE FULL VIDEO (Freepik Kling v2) ───────────────────
// Receives ALL approved shots, builds one combined narrative prompt,
// uses the first approved frame as reference, returns one single video
app.post('/api/generate-video', async (req, res) => {
  const { shots } = req.body;
  // shots = [{ imageBase64, prompt, title, index }, ...] ordered by shot index

  if (!shots || !Array.isArray(shots) || shots.length === 0) {
    return res.status(400).json({ error: 'shots array is required' });
  }

  console.log(`\n[video] Generating single full video from ${shots.length} shots...`);

  // Build one combined narrative prompt from all shots in sequence
  const combinedPrompt = shots
    .map((s, i) => `Shot ${i + 1} — ${s.title}: ${s.prompt}`)
    .join('. ');

  console.log(`[video] Combined prompt (first 300): "${combinedPrompt.slice(0, 300)}"`);

  // First approved frame anchors the character and visual style
  const firstImageBase64 = shots[0].imageBase64;

  try {
    const createRes = await fetch('https://api.freepik.com/v1/ai/image-to-video/kling-v2', {
      method: 'POST',
      headers: {
        'Content-Type': 'application/json',
        'x-freepik-api-key': FREEPIK_KEY
      },
      body: JSON.stringify({
        image:           `data:image/jpeg;base64,${firstImageBase64}`,
        prompt:          combinedPrompt,
        negative_prompt: 'blurry, low quality, distortion, watermark, jump cuts, abrupt transitions, flickering',
        duration:        '10',
        cfg_scale:       0.5
      })
    });

    console.log(`[video] Freepik create status: ${createRes.status}`);

    const createText = await createRes.text();
    if (!createRes.ok) {
      console.error(`[video] Freepik error: ${createText.slice(0, 400)}`);
      return res.status(createRes.status).json({
        error: `Freepik video API returned ${createRes.status}: ${createText.slice(0, 300)}`
      });
    }

    const createData = JSON.parse(createText);
    const taskId     = createData.data?.task_id;

    if (!taskId) {
      console.error(`[video] No task_id:`, createText.slice(0, 300));
      return res.status(500).json({ error: 'No task_id returned from Freepik video API' });
    }

    console.log(`[video] ✅ Task created: ${taskId}`);
    return res.json({ taskId });

  } catch (err) {
    console.error(`[video] ❌ ${err.message}`);
    return res.status(500).json({ error: err.message });
  }
});

// ── STAGE 3: POLL VIDEO STATUS ────────────────────────────────────────────────
app.get('/api/poll-video', async (req, res) => {
  const taskId = req.query.op;
  if (!taskId) return res.status(400).json({ error: 'op query param required' });

  console.log(`[poll] Checking task: ${taskId}`);

  try {
    const pollRes = await fetch(
      `https://api.freepik.com/v1/ai/image-to-video/kling-v2/${taskId}`,
      { headers: { 'x-freepik-api-key': FREEPIK_KEY } }
    );

    if (!pollRes.ok) {
      const errText = await pollRes.text();
      console.error(`[poll] Freepik error: ${errText.slice(0, 300)}`);
      return res.status(pollRes.status).json({
        error: `Poll returned ${pollRes.status}: ${errText.slice(0, 300)}`
      });
    }

    const pollData = await pollRes.json();
    const status   = pollData.data?.status;

    console.log(`[poll] Status: ${status}`);

    if (status === 'COMPLETED') {
      const videoUrl = pollData.data?.generated?.[0];
      if (!videoUrl) {
        return res.status(500).json({ error: 'Completed but no video URL found' });
      }
      // Fetch video and return as base64 so frontend can play + download it
      const videoBuffer = await (await fetch(videoUrl)).buffer();
      const b64Video    = videoBuffer.toString('base64');
      console.log(`[poll] ✅ Video ready`);
      return res.json({ done: true, videoBase64: b64Video });
    }

    if (status === 'FAILED') {
      return res.status(500).json({ error: 'Freepik video task failed' });
    }

    // CREATED or IN_PROGRESS — keep polling
    return res.json({ done: false, status });

  } catch (err) {
    console.error(`[poll] ❌ ${err.message}`);
    return res.status(500).json({ error: err.message });
  }
});

});

<script>
const state = {
  prompt: '',
  refImageB64: '',
  refImageMime: 'image/jpeg',
  shots: [],
  meta: {},
  frames: {},
};

// ── UPLOAD ────────────────────────────────────────────────────────────────────
function handleUpload(e) {
  const file = e.target.files[0];
  if (!file) return;
  const reader = new FileReader();
  reader.onload = ev => {
    const data = ev.target.result;
    state.refImageB64  = data.split(',')[1];
    state.refImageMime = file.type || 'image/jpeg';
    const img = document.getElementById('imgPreview');
    img.src = data;
    img.style.display = 'block';
  };
  reader.readAsDataURL(file);
}

// ── STEP 1 → 2: PARSE ────────────────────────────────────────────────────────
async function parseScript() {
  const prompt   = document.getElementById('promptInput').value.trim();
  const numShots = parseInt(document.getElementById('numShots').value) || 4;
  if (!prompt) { showError('parseError', 'Please describe your video idea.'); return; }

  state.prompt = prompt;
  document.getElementById('btnParse').disabled = true;
  clearError('parseError');
  goToStep(2);

  document.getElementById('parseStatus').innerHTML  = statusRow('spin', 'AI is reading your idea and building shot structure…');
  document.getElementById('jsonOut').innerHTML      = '';
  document.getElementById('btnGenSB').style.display = 'none';

  try {
    const res  = await fetch('/api/parse', {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({ prompt, numShots })
    });
    const data = await res.json();
    if (!res.ok) throw new Error(data.error || `Server error ${res.status}`);

    state.shots = data.shots;
    state.meta  = { title: data.title, style: data.style, character: data.character };

    document.getElementById('parseStatus').innerHTML   = '';
    document.getElementById('jsonOut').innerHTML       = highlight(JSON.stringify(data, null, 2));
    document.getElementById('btnGenSB').style.display  = 'inline-flex';
  } catch (err) {
    document.getElementById('parseStatus').innerHTML = `<div class="error-msg">${esc(err.message)}</div>`;
  }
  document.getElementById('btnParse').disabled = false;
}

// ── STEP 2 → 3: STORYBOARD ───────────────────────────────────────────────────
function goToStoryboard() {
  goToStep(3);
  buildGrid();
  generateAllFrames();
}

function buildGrid() {
  const grid = document.getElementById('sbGrid');
  grid.innerHTML = '';
  state.shots.forEach((shot, i) => {
    state.frames[i] = { b64: null, approved: false, feedbackPrompt: '' };
    grid.insertAdjacentHTML('beforeend', shotCardHTML(shot, i));
  });
  updateCounts();
}

function shotCardHTML(shot, i) {
  return `
<div class="shot-card" id="sc${i}">
  <div class="shot-frame" id="sf${i}">
    <div class="shot-num-badge">Shot ${i+1}</div>
    <div class="shot-status st-generating" id="sb${i}">Generating…</div>
    <div class="frame-overlay" id="fo${i}">
      <svg width="28" height="28" viewBox="0 0 28 28" fill="none" opacity=".35"><rect x="2" y="5" width="24" height="18" rx="3" stroke="currentColor" stroke-width="1.5"/><circle cx="10" cy="13" r="2.5" stroke="currentColor" stroke-width="1.5"/><path d="M2 20l7-7 4 4 4-5 9 8" stroke="currentColor" stroke-width="1.5" stroke-linejoin="round"/></svg>
      <span>Generating…</span>
    </div>
  </div>
  <div class="shot-info">
    <div class="shot-title">${esc(shot.title)}</div>
    <div class="shot-meta">${esc(shot.camera)} · ${shot.duration}s</div>

async function generateAllFrames() {
  for (let i = 0; i < state.shots.length; i += 2) {
    await Promise.all(
      state.shots.slice(i, i + 2).map((_, bi) => generateFrame(i + bi))
    );
  }
}

async function generateFrame(i) {
  const shot = state.shots[i];
  let prompt = shot.imagePrompt || shot.description;
  if (state.meta.character) prompt = `Character: ${state.meta.character}. ${prompt}`;
  if (state.meta.style)     prompt = `${prompt}. Visual style: ${state.meta.style}.`;
  if (state.frames[i].feedbackPrompt) prompt += ` ${state.frames[i].feedbackPrompt}`;

  setBadge(i, 'generating', 'Generating…');

  try {
    const res  = await fetch('/api/generate-image', {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({ prompt })
    });
    const data = await res.json();
    if (!res.ok) throw new Error(data.error || `Server error ${res.status}`);

    state.frames[i].b64 = data.imageBase64;
    showFrame(i, `data:${data.mimeType};base64,${data.imageBase64}`);
    setBadge(i, 'pending', 'Pending');
    setFrameActions(i);
  } catch (err) {
    setBadge(i, 'error', 'Error');
    document.getElementById(`fo${i}`).innerHTML =
      `<span style="font-size:10px;color:var(--danger);padding:8px;text-align:center;max-width:160px">${esc(err.message)}</span>`;
    document.getElementById(`sbtns${i}`).innerHTML =
      `<button class="btn btn-ghost" onclick="regenFrame(${i})">Retry</button>`;
  }
}

function showFrame(i, src) {
  const frame = document.getElementById(`sf${i}`);
  document.getElementById(`fo${i}`)?.remove();
  if (!frame.querySelector('img')) {
    const img = document.createElement('img');
    img.src = src; img.alt = `Shot ${i+1}`;
    frame.appendChild(img);
  } else {
    frame.querySelector('img').src = src;
  }
}

function setBadge(i, cls, text) {
  const b = document.getElementById(`sb${i}`);
  b.className = `shot-status st-${cls}`;
  b.textContent = text;
}

function setFrameActions(i) {
  document.getElementById(`sbtns${i}`).innerHTML = `
    <button class="btn btn-success" onclick="approveFrame(${i})">Approve</button>
    <button class="btn btn-warn"    onclick="toggleFeedback(${i})">Feedback</button>
    <button class="btn btn-ghost"   onclick="regenFrame(${i})">Regen</button>`;
}

function approveFrame(i) {
  state.frames[i].approved = true;
  document.getElementById(`sc${i}`).className = 'shot-card approved';
  setBadge(i, 'approved', 'Approved');
  document.getElementById(`fb${i}`).classList.remove('open');
  updateCounts();
}

function toggleFeedback(i) {
  document.getElementById(`fb${i}`).classList.toggle('open');
}

async function regenFrame(i) {
  const fbText = document.getElementById(`fbt${i}`)?.value?.trim() || '';
  state.frames[i].approved = false;
  state.frames[i].feedbackPrompt = fbText ? `Adjustments: ${fbText}.` : '';
  document.getElementById(`sc${i}`).className = 'shot-card';
  document.getElementById(`fb${i}`).classList.remove('open');

  const frame = document.getElementById(`sf${i}`);
  frame.querySelector('img')?.remove();
  const fo = document.createElement('div');
  fo.className = 'frame-overlay'; fo.id = `fo${i}`;
  fo.innerHTML = `<svg width="28" height="28" viewBox="0 0 28 28" fill="none" opacity=".35"><rect x="2" y="5" width="24" height="18" rx="3" stroke="currentColor" stroke-width="1.5"/></svg><span>Regenerating…</span>`;
  frame.appendChild(fo);

  updateCounts();
  await generateFrame(i);
}

function updateCounts() {
  const total    = state.shots.length;
  const approved = Object.values(state.frames).filter(f => f.approved).length;
  document.getElementById('cTotal').textContent    = total;
  document.getElementById('cApproved').textContent = approved;
  document.getElementById('cPending').textContent  = total - approved;

  const btn = document.getElementById('btnGenVid');
  btn.disabled    = approved < total;
  btn.textContent = approved < total
    ? `Generate video (${approved}/${total} approved)`
    : 'Generate video →';
}

// ── STEP 3 → 4: SINGLE FULL VIDEO GENERATION ─────────────────────────────────
async function startVideoGeneration() {
  goToStep(4);

  const statusEl  = document.getElementById('vidStatus');
  const progWrap  = document.getElementById('vidProgWrap');
  const progFill  = document.getElementById('vidProg');
  const resultsEl = document.getElementById('vidResults');
  const stripEl   = document.getElementById('storyStrip');

  progWrap.style.display = 'block';
  progFill.style.width   = '5%';
  resultsEl.innerHTML    = '';

  // Show approved storyboard thumbnails as a reference strip
  stripEl.innerHTML = state.shots.map((shot, i) => `
    <div class="strip-thumb">
      <img src="data:image/jpeg;base64,${state.frames[i].b64}" alt="Shot ${i+1}">
      <div class="strip-thumb-label">Shot ${i+1}: ${esc(shot.title)}</div>
    </div>`).join('');

  // Build shots payload — ordered by index
  const shots = state.shots.map((shot, i) => ({
    index:       i,
    title:       shot.title,
    imageBase64: state.frames[i].b64,
    prompt:      `${shot.description}. Camera: ${shot.camera}.`
  }));

  statusEl.innerHTML = statusRow('spin',
    `Submitting ${shots.length} approved storyboard frames to generate one full video…`);

  try {
    // Submit all shots as a single video generation request
    const startRes  = await fetch('/api/generate-video', {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({ shots })
    });
    const startData = await startRes.json();
    if (!startRes.ok) throw new Error(startData.error || `Server error ${startRes.status}`);

    const taskId = startData.taskId;
    progFill.style.width = '15%';

    statusEl.innerHTML = statusRow('spin',
      'Video is being generated — this takes 2–4 minutes, please keep this tab open…');

    // Poll until done
    const videoBase64 = await pollUntilDone(taskId, statusEl, progFill);

    // ── SUCCESS ──
    progFill.style.width = '100%';
    statusEl.innerHTML   = statusRow('success', 'Your video is ready!');

    resultsEl.innerHTML = `
      <div style="background:var(--surface2);border:1px solid var(--border);border-radius:var(--radius);overflow:hidden;margin-bottom:16px">
        <video
          src="data:video/mp4;base64,${videoBase64}"
          controls playsinline
          style="width:100%;display:block;background:#000;max-height:500px">
        </video>
        <div style="padding:14px 18px;display:flex;align-items:center;justify-content:space-between;gap:12px;flex-wrap:wrap">
          <div>
            <p style="font-size:13px;font-weight:500;color:var(--text);margin:0">${esc(state.meta.title || 'Frameforge Video')}</p>
            <p style="font-size:11px;color:var(--muted);font-family:'DM Mono',monospace;margin:3px 0 0">
              ${shots.length} shots · 10 seconds · generated by Kling v2
            </p>
          </div>
          <a href="data:video/mp4;base64,${videoBase64}"
             download="frameforge-${slugify(state.meta.title || 'video')}.mp4"
             class="btn btn-primary"
             style="padding:10px 24px;font-size:13px;flex-shrink:0">
            ↓ Download video
          </a>
        </div>
      </div>
      <button class="btn btn-ghost" onclick="newProject()">Start new project</button>`;

  } catch (err) {
    progFill.style.width = '0%';
    statusEl.innerHTML = `<div class="error-msg">${esc(err.message)}</div>
      <div style="margin-top:12px">
        <button class="btn btn-ghost" onclick="startVideoGeneration()">Retry</button>
        <button class="btn btn-ghost" style="margin-left:8px" onclick="goToStep(3)">← Back to storyboard</button>
      </div>`;
  }
}

async function pollUntilDone(taskId, statusEl, progFill, maxMs = 360000) {
  const t0 = Date.now();
  let progress = 15;

  while (Date.now() - t0 < maxMs) {
    await sleep(6000);

    const res  = await fetch(`/api/poll-video?op=${encodeURIComponent(taskId)}`);
    const data = await res.json();

    if (data.error) throw new Error(data.error);
    if (data.done)  return data.videoBase64;

    // Slowly nudge progress bar to show activity (caps at 90%)
    progress = Math.min(progress + 2, 90);
    progFill.style.width = `${progress}%`;

    const elapsed = Math.round((Date.now() - t0) / 1000);
    const mins    = Math.floor(elapsed / 60);
    const secs    = elapsed % 60;
    const timeStr = mins > 0 ? `${mins}m ${secs}s` : `${secs}s`;

    statusEl.innerHTML = statusRow('spin',
      `Generating full video… (${timeStr} elapsed · status: ${data.status || 'IN_PROGRESS'})`);
  }

  throw new Error('Video generation timed out after 6 minutes — please retry');
}

// ── HELPERS ───────────────────────────────────────────────────────────────────
function sleep(ms) { return new Promise(r => setTimeout(r, ms)); }

function goToStep(n) {
  [1,2,3,4].forEach(i => {
    document.getElementById(`s${i}`).className = 'section' + (i === n ? ' visible' : '');
    const si = document.getElementById(`si${i}`);
    si.className = 'step' + (i < n ? ' done' : i === n ? ' active' : '');
  });
  window.scrollTo({ top: 0, behavior: 'smooth' });
}

function newProject() {
  Object.assign(state, { prompt:'', refImageB64:'', shots:[], meta:{}, frames:{} });
  document.getElementById('promptInput').value        = '';
  document.getElementById('imgPreview').style.display = 'none';
  document.getElementById('parseError').innerHTML     = '';
  goToStep(1);
}

function showError(id, msg) { document.getElementById(id).innerHTML = `<div class="error-msg">${esc(msg)}</div>`; }
function clearError(id)     { document.getElementById(id).innerHTML = ''; }

function statusRow(type, msg) {
  const icon = type === 'spin'
    ? `<div class="spin"></div>`
    : `<div class="dot" style="background:var(--success)"></div>`;
  return `<div class="status-row">${icon}<span>${msg}</span></div>`;
}

function highlight(json) {
  return json
    .replace(/&/g,'&amp;').replace(/</g,'&lt;').replace(/>/g,'&gt;')
    .replace(/"([^"]+)":/g, '<span class="jk">"$1"</span>:')
    .replace(/: "([^"]*)"/g, ': <span class="js">"$1"</span>')
    .replace(/: (true|false|null)/g, ': <span class="jv">$1</span>')
    .replace(/: (-?\d+\.?\d*)/g, ': <span class="jn">$1</span>');
}

function esc(s) {
  return String(s||'').replace(/&/g,'&amp;').replace(/</g,'&lt;').replace(/>/g,'&gt;').replace(/"/g,'&quot;');
}

function slugify(s) {
  return String(s||'').toLowerCase().replace(/[^a-z0-9]+/g,'-').replace(/^-|-$/g,'');
}
